SELECT ?c ?value
WHERE
{
?c a ifc:IfcWindow.
?x ifc:relatedObjects_IfcRelDefines ?c.
?x ifc:relatingPropertyDefinition_IfcRelDefinesByProperties ?y.
?y ifc:hasProperties_IfcPropertySet ?z.
?z ifc:name_IfcProperty ?h.
?h express:hasString "Height".
?z ifc:nominalValue_IfcPropertySingleValue ?yy.
?yy express:hasDouble ?value.
FILTER (?value > 1)
}
